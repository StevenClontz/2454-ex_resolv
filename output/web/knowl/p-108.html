\begin{equation*} \frac{3u^2}{2} - \frac{1}{6\sqrt[3]{u^2}} = \left\langle \nabla f(\sqrt[3]{u}, u^2+1) , \Big(\frac{1}{3\sqrt[3]{u^2}},2u\Big)\right\rangle, \forall u > 0. \end{equation*}
